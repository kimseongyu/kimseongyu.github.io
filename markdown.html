def fibonacci(x):
    if x[0] == 0 :
        x[1] = x[1] + 1
        return x
    elif x[0] == 1 :
        x[2] = x[2] + 1
        return x
    else :
        y = x
        y[0] = y[0] - 1
        z = x
        z[0] = z[0] - 2
        y = fibonacci(y)
        z = fibonacci(z)
        for i in range(3):
            x[i] = y[i] + z[i]
        return x

def num(x):
    a = [x,0,0]
    fibonacci(a)
    print(a[1],a[2])
    
x = input()
x = int(x)
for i in range(x):
    y = input()
    y = int(y)
    num(y)
