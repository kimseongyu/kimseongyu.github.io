15954번 인형들

  • Algorithm
  • Java
  • Date : 2020/02/03
  • https://www.acmicpc.net/problem/15954

문제요약
서로 다른 N개의 인형에서 K개 이상의 연속되고 선택된 인형들을 선호하는 사람의 수의 표준 편차를 출력한다.

                            
public class Baekjoon15954 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        int N = input.nextInt();
        int K = input.nextInt();
        int[] array = new int[N];
        double min = 100000000;
        for (int i = 0; i < N; i++) {
            array[i] = input.nextInt();
        }
        for (int i = 0; i < N - K + 1; i++) {
            int count = K;
            while (i + count <= N) {
                double average = 0;
                double result = 0;
                for (int j = 0; j < count; j++)
                    average += array[i + j];
                average = average / count;
                for (int j = 0; j < count; j++) {
                    result += (array[i + j] - average) * (array[i + j] - average);
                }
                result = Math.sqrt(result / count);
                min = Math.min(min, result);
                count++;
            }
        }
        System.out.println(String.format("%.12f", min));
    }
}
