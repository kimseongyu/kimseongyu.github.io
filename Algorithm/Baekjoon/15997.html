15997번 승부 예측

  • Algorithm
  • Java
  • Date : 2020/02/17
  • https://www.acmicpc.net/problem/15997

문제요약
4개의 국가가 조별리그를 진행할 예정이다.
첫번째 줄에 4개 국가의 이름이 주어지고
6개의 줄에 두 국가 간의 우승, 비길, 패배 확률이 주어진다.
각 줄에 입력받은 국가가 다음 라운드에 진출할 확률을 출력하시오. 

                            
public class Baekjoon15997 {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        List list = new ArrayList();
        
        double[] score = new double[4];
        for(int i = 0; i < 4; i++)
            list.add(input.next());
        for(int i = 0; i < 6; i++) {
            String a = input.next();
            String b = input.next();
            double w = input.nextDouble();
            double d = input.nextDouble();
            double l = input.nextDouble();
            score[list.indexOf(a)] += w*3 + d*1;
            score[list.indexOf(b)] += l*3 + d*1;
        }
        double allScore = 0;
        for(int i = 0; i < 4; i++) {
            allScore += score[i];
        }
        for(int i = 0; i < 4; i++) {
            if(score[i] == 9) {
                System.out.println(1.0000000000);
            }else if(score[i] == 0) {
                System.out.println(0.0000000000);
            }else {
                System.out.println(score[i]+" "+score[i]/allScore);
            }
        }
        System.out.print(allScore);
    }
}
