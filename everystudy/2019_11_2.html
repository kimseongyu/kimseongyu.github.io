11월 12일 공부기록

  • everyday study
  • Date : 2019/11/12

컴퓨터 프로그래밍2 과제를 끝냈다.


public class StopWatchController implements Initializable{

    @FXML Label timeLabel;
    @FXML Label currentLap;
    @FXML Button startPauseButton;
    @FXML Button lapStopButton;
    
    @FXML TableView table;
    @FXML TableColumn no;
    @FXML TableColumn lapTime;
    @FXML TableColumn total;
    
    private int n;
    private long milliseconds, currentMilliseconds;
    private boolean isRunning;
    private Timeline timeline;
    private ObservableList lapTimeList;
    
    @FXML public void startPauseAction() {
        if(isRunning) {
            timeline.pause();
            isRunning = false;
            startPauseButton.setText("시작");
            lapStopButton.setText("초기화");
        }else {
            timeline.play();
            isRunning = true;
            startPauseButton.setText("일시정지");
            lapStopButton.setText("랩타임");
            lapStopButton.setDisable(false);
        }
    }
    @FXML public void labStopAction() {
        if(isRunning) {
            lapTimeList.add(new LapTable(n++,currentLap.getText(),timeLabel.getText()));
            table.scrollTo(lapTimeList.size());
        }else {
            startPauseButton.setText("시작");
            lapStopButton.setText("랩타임");
            lapStopButton.setDisable(true);
            timeline.pause();
            timeLabel.setText("00:00:00.000");
            milliseconds = 0;
            n = 0;
            lapTimeList.clear();
        }
        currentLap.setText(millisecondsToTime(0));
        currentMilliseconds = 0;
    }
    @Override
    public void initialize(URL url, ResourceBundle resource) {
        lapTimeList = FXCollections.observableArrayList();
        table.setItems(lapTimeList);
        table.setPlaceholder(new Label("랩 데이터가 없음"));
        
        no.setCellValueFactory(param -> param.getValue().getNoProperty().asObject());
        lapTime.setCellValueFactory(param -> param.getValue().getLapTimeProperty());
        total.setCellValueFactory(param -> param.getValue().getTotalProperty());
        
        KeyFrame keyFrame = new KeyFrame(Duration.ONE, new EventHandler() {
            @Override
            public void handle(ActionEvent event) {
                timeLabel.setText(millisecondsToTime(milliseconds++));
                currentLap.setText(millisecondsToTime(currentMilliseconds++));
            }
        });
        
        timeline = new Timeline(keyFrame);
        timeline.setCycleCount(Timeline.INDEFINITE);
        timeline.setAutoReverse(false);
    }
    private String millisecondsToTime(long milliseconds) {
        int h = 0, m = 0, s = 0, ms = 0;
        ms = (int)(milliseconds % 1000);
        milliseconds = milliseconds/1000;
        s = (int)(milliseconds % 60);
        milliseconds = milliseconds/60;
        m = (int)(milliseconds % 60);
        milliseconds = milliseconds/60;
        h = (int)milliseconds;
        
        return String.format("%02d:%02d:%02d.%03d", h, m, s, ms);
    }

}
