public class TransferRunnable extends Thread{
    Bank bank;
    int fromAccount;
    double maxAmount;
    final int DELAY = 10;
    
    public TransferRunnable(Bank bank,int fromAccount, double maxAmount) {
        this.bank = bank;
        this.fromAccount = fromAccount;
        this.maxAmount = maxAmount;
    }
    
    @Override
    public void run() {
        while(true) {
        bank.transfer(fromAccount, (int)(Math.random()*100), (int)(Math.random()*maxAmount));
        try {Thread.sleep(DELAY);}catch(Exception e) {}
        }
    }
}

public class SynchBankTest {
    public static final int NCUSTOMERS = 100;
    public static final int INITIAL_BALANCE = 1000;
    public static void main(String[] args) {
        int i, count;
        Bank bank = new Bank(NCUSTOMERS);
        
        for(i = 0; i < NCUSTOMERS; i++) {
            bank.customer[i] = new Customer("CNU" + Integer.toString(i));
            bank.customer[i].addAccount(new CheckingAccount(INITIAL_BALANCE));
        }
        Thread[] thread = new Thread[NCUSTOMERS];
        for(i = 0; i < NCUSTOMERS; i++) {
            thread[i] = new Thread(new TransferRunnable(bank, i,INITIAL_BALANCE));
            thread[i].start();
        }
    }
}
