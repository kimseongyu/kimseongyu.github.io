public static void radixSort(int[] a) {
    int[][] list = new int[10][a.length];
    int[] size = new int[10];
    int index;
    int max = 0;
    int n = 1;
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < a.length; j++) {
            index = (a[j]%(int)Math.pow(10, n))/(int)Math.pow(10,n-1);
            list[index][size[index]] = a[j];
            if(max < j) {
                max = j;
            }
        }
        
        if(max/Math.pow(10,n) > 0) {
            n++;
        }
    }
}

public class SortTest {
    public static int[] makeRandArr(int n, int low, int high) {
        int[] list = new int[n];
        int number;
        for(int i = 0; i < n;) {
            number = (int)(Math.random()*high);
            if(number >= low) {
                list[i] = number;
                i++;
            }
        }
        return list;
    }
